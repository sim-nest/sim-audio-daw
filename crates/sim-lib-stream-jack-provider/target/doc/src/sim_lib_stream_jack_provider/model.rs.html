3use std::sync::Arc;
4
5use sim_kernel::{
6    AbiVersion, Export, Lib, LibManifest, LibTarget, Linker, LoadCx, Result, Symbol, Version,

8use sim_lib_stream_host::{

12/// Returns the provider library identity.
13pub fn jack_provider_symbol() -> Symbol {
14    Symbol::qualified("audio/provider", "jack")
15}
16
17/// Returns the backend identity used by native JACK streams.
18pub fn jack_backend_symbol() -> Symbol {
19    Symbol::qualified("stream/host", "jack")
20}
21
22/// Returns the modeled JACK audio site export symbol.
23pub fn jack_modeled_site_symbol() -> Symbol {
24    Symbol::qualified("audio/site", "jack-modeled")
25}
26
27/// Builds the deterministic modeled JACK site used by default validation.
28pub fn default_modeled_jack_site() -> Arc<dyn AudioSite> {
29    let key = AudioSiteKey(jack_modeled_site_symbol());
30    let card = AudioDeviceCard::modeled(key, "JACK Provider Modeled");
31    Arc::new(ModeledAudioSite::new(card, Arc::new(FakeBackend::new())))
32}
33
34/// Enumerates JACK provider sites for the active feature set.
35#[cfg(not(feature = "jack-hardware"))]
36pub fn enumerate_jack_sites() -> Result<Vec<Arc<dyn AudioSite>>> {
37    Ok(vec![default_modeled_jack_site()])
38}
39
40/// Enumerates JACK provider sites for the active feature set.
41#[cfg(feature = "jack-hardware")]
42pub fn enumerate_jack_sites() -> Result<Vec<Arc<dyn AudioSite>>> {
43    crate::native::enumerate_jack_hardware_sites()
44}

46/// FFI-free modeled JACK provider library used by the host loader tests.
47pub struct JackProviderModeled;
48
49impl Lib for JackProviderModeled {
50    fn manifest(&self) -> LibManifest {
51        LibManifest {
52            id: jack_provider_symbol(),
53            version: Version("0.1.0".to_owned()),
54            abi: AbiVersion { major: 0, minor: 1 },
55            target: LibTarget::HostRegistered,
56            requires: Vec::new(),
57            capabilities: Vec::new(),
58            exports: vec![Export::Site {
59                symbol: jack_modeled_site_symbol(),
60                runtime_id: None,
61            }],
62        }
63    }
64
65    fn load(&self, _cx: &mut LoadCx, _linker: &mut Linker) -> Result<()> {
66        Ok(())
67    }
